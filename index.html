async function send() { var text = dom.input.value.trim(); if (!text || state.busy) return; if (typeof puter === "undefined" || !puter.ai) { toast("AI not ready yet","warning"); return; } if (!state.data) { toast("No data loaded","error"); return; } state.busy = true; dom.sendBtn.disabled = true; dom.input.value = ""; dom.input.style.height = "auto"; addMsg("user", text); state.history.push({role:"user", content:text}); // Get ALL content from selected category/video var chunks = search(text); // ALSO get extra broad context - grab more chunks var broadChunks = searchBroad(text); // Combine and deduplicate var allChunks = mergeChunks(chunks, broadChunks); // Build full transcript context var ctx = buildContext(allChunks); var sysPrompt = "You are an expert AI assistant with deep knowledge of the provided transcripts.\n\n" + "YOUR JOB:\n" + "- Answer ANY question the user asks using the transcript content\n" + "- Answer from ANY angle - direct, indirect, summary, detail, comparison, explanation\n" + "- If the exact words aren't there but the concept is - STILL ANSWER using what you know from the transcripts\n" + "- Combine information from multiple parts of the transcript to form complete answers\n" + "- Explain, summarize, compare, list, or elaborate based on what's in the transcripts\n" + "- If asked something not covered at all, say what IS covered and offer to help with that\n" + "- NEVER say 'I cannot answer' if relevant content exists anywhere in the transcripts\n" + "- Be thorough, helpful, and conversational\n\n" + "TRANSCRIPT CONTENT:\n" + ctx; var msgs = [{role:"system", content:sysPrompt}]; var hist = state.history.slice(-10); for (var i = 0; i < hist.length; i++) msgs.push(hist[i]); var model = dom.modelPicker.value || "gpt-4o-mini"; var botEl = addMsg("bot","",true); var bubEl = botEl.querySelector(".msg-bubble"); var footEl = botEl.querySelector(".msg-footer"); var full = ""; try { var stream = await puter.ai.chat(msgs, {model:model, stream:true}); bubEl.innerHTML = ""; bubEl.classList.add("streaming"); for await (var part of stream) { var token = (part && part.text) ? part.text : ""; full += token; bubEl.innerHTML = fmt(full); scrollDown(); } bubEl.classList.remove("streaming"); if (allChunks.length > 0) { var seen = {}; allChunks.slice(0,3).forEach(function(c) { var key = formatCat(c.cat) + " > " + c.file.replace(/\.(txt|pdf)$/i,""); if (!seen[key]) { seen[key] = true; var tag = document.createElement("span"); tag.className = "source-tag"; tag.textContent = "\uD83D\uDCC4 " + key; footEl.appendChild(tag); } }); } state.history.push({role:"assistant", content:full}); } catch(err) { bubEl.classList.remove("streaming"); bubEl.innerHTML = '\u26A0 ' + (err.message||"Error") + ''; } state.busy = false; dom.sendBtn.disabled = false; dom.input.focus(); }
